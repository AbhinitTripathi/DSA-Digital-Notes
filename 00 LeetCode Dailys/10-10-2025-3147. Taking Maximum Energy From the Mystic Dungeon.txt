class Solution {
public:
    int maximumEnergy(vector<int>& energy, int k) {
        int n = energy.size();
        int max_sum = -1e9;
        vector<int> dp(n, 0);
        for (int i = n-1; i >= 0; --i) {
            dp[i] = energy[i];
            if (i + k < n)
                dp[i] += dp[i + k];
        }

        for (int i = 0; i < n; ++i) {
            max_sum = max(max_sum, dp[i]);
        }

        return max_sum;
    }
};
