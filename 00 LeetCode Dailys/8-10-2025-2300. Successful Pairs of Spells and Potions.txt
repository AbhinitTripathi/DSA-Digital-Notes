class Solution {
public:
    vector<int> successfulPairs(vector<int>& s, vector<int>& p, long long success) {
        int n = s.size(), m = p.size();
        
        sort(p.begin(), p.end());
        vector<int> res;
        for (int i = 0; i < n; ++i) {
            long long lb_of = ceil((double) success / s[i]);
            int lb_ind = lower_bound(p.begin(), p.end(), lb_of) - p.begin();
            res.push_back(m - lb_ind);
        }

        return res;
    }
};
