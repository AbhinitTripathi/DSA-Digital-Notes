class Solution {
private:
    bool isAnagram(string a, string b) {
        if (a.size() != b.size()) return false;
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        return a == b;
    }
public:
    vector<string> removeAnagrams(vector<string>& words) {
        vector<string> result;
        
        for (auto& word : words) {
            if (!result.empty() && isAnagram(result.back(), word)) {
                continue;
            }
            result.push_back(word);
        }
        
        return result;
    }
};
