class Solution {
public:
    int maxFrequencyElements(vector<int>& nums) {
        vector<int> freq(101, 0);
        int max_freq = 0;
        int max_freq_els = 0;
        for (int x : nums) {
            freq[x]++;
            if (freq[x] > max_freq) {
                max_freq = freq[x];
                max_freq_els = 1;
            } else if (freq[x] == max_freq) {
                max_freq_els++;
            }
        }

        return max_freq * max_freq_els;
    }
};
