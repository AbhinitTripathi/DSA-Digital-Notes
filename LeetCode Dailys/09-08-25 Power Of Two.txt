----------------------------- BRUTE -----------------------------
class Solution {
public:
    bool isPowerOfTwo(int n) {
        if(n <= 0) return false;
        while (n > 1) {
            if(n % 2 == 0) n /= 2;
            else return false;
        }
        return true;
    }
};

----------------------------- OPTIMAL -----------------------------
class Solution {
public:
    bool isPowerOfTwo(int n) {
        if(n <= 0) return false;
        return (n & n-1) == 0;
    }
};
